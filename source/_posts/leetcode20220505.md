# 13. 罗马数字转整数

比较直接的解法，一个指针指当前位，另一个指针指下一位。如果当前位和下一位凑成了4、9等数，就在结果中加上，然后当前指针指向下下位。如果当前位和下一位没有凑成特殊情况，就在结果中加上当前位的数，并把当前位指针指向下一位。

```python
class Solution:
    def romanToInt(self, s: str) -> int:
        res = 0
        s_len = len(s)
        index = 0
        while index<s_len:
            if s[index] == 'I':
                next_index = index+1
                if next_index < s_len:
                    if s[next_index] == 'V':
                        res += 4
                        index += 2
                    elif s[next_index] == 'X':
                        res += 9
                        index += 2
                    else:
                        res += 1
                        index += 1
                else:
                    res += 1
                    index += 1
            elif s[index] == 'X':
                next_index = index+1
                if next_index < s_len:
                    if s[next_index] == 'L':
                        res += 40
                        index += 2
                    elif s[next_index] == 'C':
                        res += 90
                        index += 2
                    else:
                        res += 10
                        index += 1
                else:
                    res += 10
                    index += 1
            elif s[index] == 'C':
                next_index = index+1
                if next_index < s_len:
                    if s[next_index] == 'D':
                        res += 400
                        index += 2
                    elif s[next_index] == 'M':
                        res += 900
                        index += 2
                    else:
                        res += 100
                        index += 1
                else:
                    res += 100
                    index += 1
            elif s[index] == 'V':
                res += 5
                index += 1
            elif s[index] == 'L':
                res += 50
                index += 1
            elif s[index] == 'D':
                res += 500
                index += 1
            elif s[index] == 'M':
                res += 1000
                index += 1

        return res
```
