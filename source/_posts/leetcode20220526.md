# 206. 反转链表

常见的用栈反转。

```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: ListNode) -> ListNode:
        stack = []
        node = head
        while node:
            stack.append(node.val)
            node = node.next
        res = ListNode()
        node = res
        while stack:
            node.next = ListNode(stack.pop())
            node = node.next
        return res.next
```
